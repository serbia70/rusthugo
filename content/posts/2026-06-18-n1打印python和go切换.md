创建服务
```
cat > /etc/systemd/system/printer-n1.service << 'EOF'
[Unit]
Description=N1 Printer Service (Go)
After=network.target network-online.target
Wants=network-online.target

[Service]
Type=simple
User=root
WorkingDirectory=/opt/printer-n1
ExecStart=/opt/printer-n1/printer-arm64 -config /opt/printer-n1/config.json
Restart=always
RestartSec=10
Environment=LANG=zh_CN.UTF-8
Environment=LC_ALL=zh_CN.UTF-8

[Install]
WantedBy=multi-user.target
EOF
```
