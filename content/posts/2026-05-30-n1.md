# 打印
## 加载 USB 打印机内核模块
modprobe usblp

## 确认设备文件出现了
ls -la /dev/usb/lp*

## 让模块开机自动加载
echo usblp >> /etc/modules

## 重启容器，让它重新发现打印机
docker compose restart app

## 检查日志，确认发现打印机
docker logs food-server 2>&1 | grep -i "printer\|发现"

## 日志排查

docker compose logs --tail 50


# 安装docker
sudo apt update && sudo apt upgrade -y
