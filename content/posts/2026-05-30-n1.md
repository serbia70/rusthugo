# 打印
## 加载 USB 打印机内核模块
modprobe usblp

## 确认设备文件出现了
ls -la /dev/usb/lp*

## 让模块开机自动加载
echo usblp >> /etc/modules

## 重启容器，让它重新发现打印机
docker compose restart app

## 检查日志，确认发现打印机
docker logs food-server 2>&1 | grep -i "printer\|发现"

## 重启 N1 后打印机驱动自动加载，Docker 容器自动运行，不需要手动操作

echo "usblp" >> /etc/modules-load.d/printer.conf && systemctl enable docker

## 日志排查

docker compose logs --tail 50

echo 'deb [signed-by=/usr/share/keyrings/cloudflare-public-v2.gpg] https://pkg.cloudflare.com/cloudflared any main' | sudo tee /etc/apt/sources.list.d/cloudflared.list

## 连接wifi

### 常用命令

查看wifi

nmcli device wifi list

假设您的 SSID 是 MyHomeWiFi，密码是 MyPassword123

nmcli device wifi connect MyHomeWiFi password MyPassword123

图形

sudo nmtui

windows查看wifi

netsh wlan show networks

查询U盘盘符

lsblk

假设它是 /dev/sdb2

挂载
、、、
mount /dev/sdb2 /mnt
、、、


直接创建 WiFi 配置文件。NetworkManager 是 Armbian 默认的管理工具，我们直接给它塞一个连接文件
、、、
nano /mnt/etc/NetworkManager/system-connections/serbia70.nmconnection
、、、

、、、
[connection]
id=serbia70
type=wifi

[wifi]
mode=infrastructure
ssid=这里填你的wifi

[wifi-security]
auth-alg=open
key-mgmt=wpa-psk
psk=这里填你的wifi密码

[ipv4]
method=auto

[ipv6]
addr-gen-mode=stable-privacy
method=auto
、、、

设置权限
、、、
chmod 600 /mnt/etc/NetworkManager/system-connections/Tech_D0021161.nmconnection
、、、

卸载并把U盘插到N1盒子

umount /mnt

扫描

、、、
nmcli device wifi list
、、、

写入
、、、
nmcli device wifi connect "WiFi名称" password "WiFi密码"
、、、

ip

netstat -ano | findstr ":22"
